ingredients:
  - item: Butter
    amount: 100g
  - item: White Sugar
    amount: 100g
  - item: Brown Sugar
    amount: 100g
  - item: Eggs
    amount: 1
  - item: Vanilla (extract)
    amount: 1tsp
  - item: Baking Soda
    amount: 0.5tsp
  - item: Water
    amount: 1tsp
  - item: Salt
    amount: 0.25tsp
  - item: Flour
    amount: 200g
  - item: Chocolate Chips
    amount: 200g


directions:
  - Pour the ingredients into a bowl.
  - Mix until smooth.
  - Mold spheres and place them on a pan.<br>the batter will sink into a circular cookie shape.
  - Cook for 10min at 175 °C
  - Cause the Grandma-pocalapse again.
